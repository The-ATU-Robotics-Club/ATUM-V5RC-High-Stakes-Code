81 bool canAccept(const Unit &error) {

83 timeoutTimer = Timer{timeout};

85 const second_t currentTime{time()};
86 const UnitsPerSecond deriv{(error - prevError) / (currentTime - prevTime)};
87 accepted = abs(error) <= maxError;
