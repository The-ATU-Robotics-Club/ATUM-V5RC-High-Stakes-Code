158 double getPositionOutput(const Unit state, const Unit reference) const {
159 if(!positionController || abs(end - state) >= positionControlThreshold) {
160 return 0.0;
161 }
162 const double positionError{positionControlThreshold >= Unit{infinite} ?
163 difference(reference, state) :
164 difference(end, state)};
165 return positionController->getOutput(positionError);
166 }

175 double getAccelerationOutput(const UnitsPerSecondSq accel) const {
176 double accelerationOutput{getValueAs<UnitsPerSecondSq>(accel)};
177 if(accel < UnitsPerSecondSq{0}) {
178 accelerationOutput *= profile.isBackwards() ? kA.accel : kA.decel;
179 } else {
180 accelerationOutput *= profile.isBackwards() ? kA.decel : kA.accel;
181 }
182 return accelerationOutput;
183 }
