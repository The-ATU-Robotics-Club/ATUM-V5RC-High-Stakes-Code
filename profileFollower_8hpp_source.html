164 double getAccelerationOutput(const UnitsPerSecondSq accel) const {
165 double accelerationOutput{getValueAs<UnitsPerSecondSq>(accel)};
166 if(accel < UnitsPerSecondSq{0}) {
167 accelerationOutput *= profile.isBackwards() ? kA.accel : kA.decel;
168 } else {
169 accelerationOutput *= profile.isBackwards() ? kA.decel : kA.accel;
170 }
171 return accelerationOutput;
172 }
