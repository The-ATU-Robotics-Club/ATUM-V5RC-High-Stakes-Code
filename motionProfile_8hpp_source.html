246 Point getClosestPoint(const Unit s) {
247 // Return appropriate value if behind start or in front of end.
248 if(distance(s, start) > Unit{0}) {
249 return {start,
250 UnitsPerSecond{0},
251 UnitsPerSecondSq{0},
252 UnitsPerSecondCb{(isBackwards()) ? -params.maxJ : params.maxJ},
253 0_s};
254 } else if(distance(s, end) < Unit{0}) {
255 return {end,
256 UnitsPerSecond{0},
257 UnitsPerSecondSq{0},
258 UnitsPerSecondCb{0},
259 points[6].t};
260 }
261 const auto [t0, t2] = getTimeBounds(s);
262 const second_t t1{searchForClosestPointTime(s, t0, t2)};
263 return getPointAt(t1);
264 }

272 Point getPointAt(const second_t t) {
273 if(t < 0_s) {
274 return {start,
275 UnitsPerSecond{0},
276 UnitsPerSecondSq{0},
277 UnitsPerSecondCb{(isBackwards()) ? -params.maxJ : params.maxJ},
278 0_s};
279 } else if(t >= points[6].t) {
280 return {end,
281 UnitsPerSecond{0},
282 UnitsPerSecondSq{0},
283 UnitsPerSecondCb{0},
284 points[6].t};
285 }
286 int i{0};
287 while(t >= points[i].t) {
288 i++;
289 }
290 second_t dt{t};
291 if(i > 0) {
292 dt -= points[i - 1].t;
293 }
294 Point p;
295 p.s = UnitKinematics::position(
296 dt, points[i].s, points[i].v, points[i].a, points[i].j);
297 p.v = UnitKinematics::velocity(dt, points[i].v, points[i].a, points[i].j);
298 p.a = points[i].a + points[i].j * dt;
299 p.j = points[i].j;
300 p.t = t;
301 return p;
302 }

310 std::pair<second_t, second_t> getTimeBounds(const Unit s) {
311 // Finds first point we're past (starts with 1 since we know we're past the
312 // start if this ever gets called).
313 int i{1};
314 while(i < 7 && distance(s, points[i].s) < Unit{0}) {
315 i++;
316 }
317 if(i < 2) {
318 return {0_s, points[0].t};
319 } else if(i >= 7) {
320 return {points[5].t, points[6].t};
321 } else {
322 return {points[i - 2].t, points[i - 1].t};
323 }
324 }

335 second_t searchForClosestPointTime(const Unit s, second_t t0, second_t t2) {
336 for(std::size_t n{0}; n < params.searchIterations; n++) {
337 const second_t t1{(t0 + t2) / 2.0};
338 if(distance(s, getPointAt(t1).s) >= Unit{0}) {
339 t2 = t1;
340 } else {
341 t0 = t1;
342 }
343 }
344 return (t0 + t2) / 2.0;
345 }

368 void beginProfile() {
369 const bool reachesMaxAccel{params.maxA * params.maxA / params.maxJ <
370 params.maxV};
371 second_t jerkT{params.maxA / params.maxJ};
372 if(!reachesMaxAccel) {
373 jerkT = second_t{sqrt(getValueAs<second_t>(jerkT))};
374 }
375 const Unit jerkDistance{params.maxJ * 2.0 * pow<3>(jerkT)};
376 Unit fullDistance{jerkDistance};
377 if(reachesMaxAccel) {
378 const second_t constantAccelT =
379 (params.maxV - params.maxJ * jerkT * jerkT) / params.maxA;
380 fullDistance = 2.0 * params.maxV * jerkT +
381 params.maxJ * jerkT * jerkT * constantAccelT +
382 params.maxA * constantAccelT * constantAccelT;
383 }
384 const Unit absTarget{abs(target)};
385 if(absTarget < jerkDistance) {
386 profile4Stage();
387 } else if(!reachesMaxAccel) {
388 profile5Stage();
389 } else if(absTarget < fullDistance) {
390 profile6Stage();
391 } else {
392 profileAllStages();
393 }
394 }

400 void profile4Stage() {
401 logger.debug("Generated profile is 4 stages.");
402 const Unit absTarget{abs(target)};
403 points[0].t = points[2].t = points[4].t = points[6].t =
404 second_t{cbrt(getValueAs<Unit>(absTarget) /
405 getValueAs<UnitsPerSecondCb>(params.maxJ) / 2.0)};
406 points[1].s = points[0].t * points[0].t * points[0].t * params.maxJ / 6.0;
407 points[3].s = points[5].s = 0.5 * absTarget - points[1].s;
408 points[1].v = points[2].v = points[5].v = points[6].v =
409 0.5 * params.maxJ * points[0].t * points[0].t;
410 points[3].v = points[4].v = points[1].v * 2.0;
411 points[1].a = points[2].a = params.maxJ * points[0].t;
412 points[5].a = points[6].a = -points[1].a;
413 }

419 void profile5Stage() {
420 logger.debug("Generated profile is 5 stages.");
421 const Unit absTarget{abs(target)};
422 points[0].t = points[2].t = points[4].t = points[6].t =
423 sqrt(params.maxV / params.maxJ);
424 points[3].t = (absTarget - params.maxV * points[0].t * 2.0) / params.maxV;
425 points[1].s = points[0].t * points[0].t * points[0].t * params.maxJ / 6.0;
426 points[3].s = points[5].s = params.maxV * points[0].t - points[1].s;
427 points[4].s = absTarget - points[1].s * 2.0 - points[3].s * 2.0;
428 points[1].v = points[2].v = points[5].v = points[6].v = params.maxV / 2.0;
429 points[3].v = points[4].v = params.maxV;
430 points[1].a = points[2].a = params.maxJ * points[0].t;
431 points[5].a = points[6].a = -points[1].a;
432 }

438 void profile6Stage() {
439 logger.debug("Generated profile is 6 stages.");
440 const Unit absTarget{abs(target)};
441 const UnitsPerSecond b{3.0 * params.maxA * params.maxA / params.maxJ};
442 const Unit c{2.0 * params.maxA * params.maxA * params.maxA / params.maxJ /
443 params.maxJ -
444 absTarget};
445 const auto discriminant = sqrt(b * b - 4.0 * params.maxA * c);
446 const second_t t{max((-b + discriminant) / (2.0 * params.maxA),
447 (-b - discriminant) / (2.0 * params.maxA))};
448 points[0].t = points[2].t = points[4].t = points[6].t =
449 params.maxA / params.maxJ;
450 points[1].t = points[5].t = t;
451 points[1].a = points[2].a = params.maxJ * points[0].t;
452 points[5].a = points[6].a = -points[1].a;
453 points[1].v = points[6].v = 0.5 * params.maxJ * points[0].t * points[0].t;
454 points[2].v = points[5].v = points[1].v + params.maxA * points[1].t;
455 points[3].v = points[4].v = points[2].v + points[2].a * points[2].t -
456 0.5 * params.maxJ * points[2].t * points[2].t;
457 points[1].s = params.maxJ * points[0].t * points[0].t * points[0].t / 6.0;
458 points[2].s = points[6].s = points[1].v * points[1].t +
459 0.5 * points[1].a * points[1].t * points[1].t;
460 points[3].s = points[5].s =
461 points[2].v * points[2].t +
462 0.5 * points[2].a * points[2].t * points[2].t -
463 params.maxJ * points[2].t * points[2].t * points[2].t / 6.0;
464 }

470 void profileAllStages() {
471 logger.debug("Generated profile is 7 (all) stages.");
472 const Unit absTarget{abs(target)};
473 points[0].t = points[2].t = points[4].t = points[6].t =
474 params.maxA / params.maxJ;
475 points[1].s = params.maxJ * points[0].t * points[0].t * points[0].t / 6.0;
476 points[1].v = points[6].v = 0.5 * params.maxJ * points[0].t * points[0].t;
477 points[1].a = points[2].a = params.maxA;
478 points[1].t = points[5].t = (params.maxV - points[1].v * 2.0) / params.maxA;
479 points[2].s = points[6].s = points[1].v * points[1].t +
480 0.5 * points[1].a * points[1].t * points[1].t;
481 points[2].v = points[5].v = points[1].v + points[1].a * points[1].t;
482 points[3].s = points[5].s =
483 points[2].v * points[2].t +
484 0.5 * points[2].a * points[2].t * points[2].t -
485 params.maxJ * points[2].t * points[2].t * points[2].t / 6.0;
486 points[3].v = points[4].v = params.maxV;
487 points[3].t =
488 (absTarget - 2.0 * (points[1].s + points[2].s + points[3].s)) /
489 params.maxV;
490 points[4].s = points[3].v * points[3].t;
491 points[5].a = points[6].a = -params.maxA;
492 }

499 void finishProfile() {
500 points[0].j = points[6].j = params.maxJ;
501 points[2].j = points[4].j = -params.maxJ;
502 if(target < Unit{0.0}) {
503 for(Point &p : points) {
504 p.s *= -1;
505 p.v *= -1;
506 p.a *= -1;
507 p.j *= -1;
508 }
509 }
510 points[0].s = start;
511 for(std::size_t i{1}; i < 7; i++) {
512 points[i].t += points[i - 1].t;
513 points[i].s += points[i - 1].s;
514 }
515 }
