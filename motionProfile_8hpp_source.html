237 Point getClosestPoint(const Unit s) {
238 // Return appropriate value if behind start or in front of end.
239 if(distance(s, start) > Unit{0}) {
240 return {start,
241 UnitsPerSecond{0},
242 UnitsPerSecondSq{0},
243 UnitsPerSecondCb{(isBackwards()) ? -params.maxJ : params.maxJ},
244 0_s};
245 } else if(distance(s, end) < Unit{0}) {
246 return {end,
247 UnitsPerSecond{0},
248 UnitsPerSecondSq{0},
249 UnitsPerSecondCb{0},
250 points[6].t};
251 }
252 const auto [t0, t2] = getTimeBounds(s);
253 const second_t t1{searchForClosestPointTime(s, t0, t2)};
254 return getPointAt(t1);
255 }

263 Point getPointAt(const second_t t) {
264 if(t < 0_s) {
265 return {start,
266 UnitsPerSecond{0},
267 UnitsPerSecondSq{0},
268 UnitsPerSecondCb{(isBackwards()) ? -params.maxJ : params.maxJ},
269 0_s};
270 } else if(t >= points[6].t) {
271 return {end,
272 UnitsPerSecond{0},
273 UnitsPerSecondSq{0},
274 UnitsPerSecondCb{0},
275 points[6].t};
276 }
277 int i{0};
278 while(t >= points[i].t) {
279 i++;
280 }
281 second_t dt{t};
282 if(i > 0) {
283 dt -= points[i - 1].t;
284 }
285 Point p;
286 p.s = UnitKinematics::position(
287 dt, points[i].s, points[i].v, points[i].a, points[i].j);
288 p.v = UnitKinematics::velocity(dt, points[i].v, points[i].a, points[i].j);
289 p.a = points[i].a + points[i].j * dt;
290 p.j = points[i].j;
291 p.t = t;
292 return p;
293 }

301 std::pair<second_t, second_t> getTimeBounds(const Unit s) {
302 // Finds first point we're past (starts with 1 since we know we're past the
303 // start if this ever gets called).
304 int i{1};
305 while(i < 7 && distance(s, points[i].s) < Unit{0}) {
306 i++;
307 }
308 if(i < 2) {
309 return {0_s, points[0].t};
310 } else if(i >= 7) {
311 return {points[5].t, points[6].t};
312 } else {
313 return {points[i - 2].t, points[i - 1].t};
314 }

326 second_t searchForClosestPointTime(const Unit s, second_t t0, second_t t2) {
327 for(std::size_t n{0}; n < params.searchIterations; n++) {
328 const second_t t1{(t0 + t2) / 2.0};
329 if(distance(s, getPointAt(t1).s) >= Unit{0}) {
330 t2 = t1;
331 } else {
332 t0 = t1;
333 }
334 }
335 return (t0 + t2) / 2.0;
336 }

359 void beginProfile() {
360 const bool reachesMaxAccel{params.maxA * params.maxA / params.maxJ <
361 params.maxV};
362 second_t jerkT{params.maxA / params.maxJ};
363 if(!reachesMaxAccel) {
364 jerkT = second_t{sqrt(getValueAs<second_t>(jerkT))};
365 }
366 const Unit jerkDistance{params.maxJ * 2.0 * pow<3>(jerkT)};
367 Unit fullDistance{jerkDistance};
368 if(reachesMaxAccel) {
369 const second_t constantAccelT =
370 (params.maxV - params.maxJ * jerkT * jerkT) / params.maxA;
371 fullDistance = 2.0 * params.maxV * jerkT +
372 params.maxJ * jerkT * jerkT * constantAccelT +
373 params.maxA * constantAccelT * constantAccelT;
374 }
375 const Unit absTarget{abs(target)};
376 if(absTarget < jerkDistance) {
377 profile4Stage();
378 } else if(!reachesMaxAccel) {
379 profile5Stage();
380 } else if(absTarget < fullDistance) {
381 profile6Stage();
382 } else {
383 profileAllStages();
384 }
385 }

391 void profile4Stage() {
392 logger.debug("Generated profile is 4 stages.");
393 const Unit absTarget{abs(target)};
394 points[0].t = points[2].t = points[4].t = points[6].t =
395 second_t{cbrt(getValueAs<Unit>(absTarget) /
396 getValueAs<UnitsPerSecondCb>(params.maxJ) / 2.0)};
397 points[1].s = points[0].t * points[0].t * points[0].t * params.maxJ / 6.0;
398 points[3].s = points[5].s = 0.5 * absTarget - points[1].s;
399 points[1].v = points[2].v = points[5].v = points[6].v =
400 0.5 * params.maxJ * points[0].t * points[0].t;
401 points[3].v = points[4].v = points[1].v * 2.0;
402 points[1].a = points[2].a = params.maxJ * points[0].t;
403 points[5].a = points[6].a = -points[1].a;
404 }

410 void profile5Stage() {
411 logger.debug("Generated profile is 5 stages.");
412 const Unit absTarget{abs(target)};
413 points[0].t = points[2].t = points[4].t = points[6].t =
414 sqrt(params.maxV / params.maxJ);
415 points[3].t = (absTarget - params.maxV * points[0].t * 2.0) / params.maxV;
416 points[1].s = points[0].t * points[0].t * points[0].t * params.maxJ / 6.0;
417 points[3].s = points[5].s = params.maxV * points[0].t - points[1].s;
418 points[4].s = absTarget - points[1].s * 2.0 - points[3].s * 2.0;
419 points[1].v = points[2].v = points[5].v = points[6].v = params.maxV / 2.0;
420 points[3].v = points[4].v = params.maxV;
421 points[1].a = points[2].a = params.maxJ * points[0].t;
422 points[5].a = points[6].a = -points[1].a;
423 }

429 void profile6Stage() {
430 logger.debug("Generated profile is 6 stages.");
431 const Unit absTarget{abs(target)};
432 const UnitsPerSecond b{3.0 * params.maxA * params.maxA / params.maxJ};
433 const Unit c{2.0 * params.maxA * params.maxA * params.maxA / params.maxJ /
434 params.maxJ -
435 absTarget};
436 const auto discriminant = sqrt(b * b - 4.0 * params.maxA * c);
437 const second_t t{max((-b + discriminant) / (2.0 * params.maxA),
438 (-b - discriminant) / (2.0 * params.maxA))};
439 points[0].t = points[2].t = points[4].t = points[6].t =
440 params.maxA / params.maxJ;
441 points[1].t = points[5].t = t;
442 points[1].a = points[2].a = params.maxJ * points[0].t;
443 points[5].a = points[6].a = -points[1].a;
444 points[1].v = points[6].v = 0.5 * params.maxJ * points[0].t * points[0].t;
445 points[2].v = points[5].v = points[1].v + params.maxA * points[1].t;
446 points[3].v = points[4].v = points[2].v + points[2].a * points[2].t -
447 0.5 * params.maxJ * points[2].t * points[2].t;
448 points[1].s = params.maxJ * points[0].t * points[0].t * points[0].t / 6.0;
449 points[2].s = points[6].s = points[1].v * points[1].t +
450 0.5 * points[1].a * points[1].t * points[1].t;
451 points[3].s = points[5].s =
452 points[2].v * points[2].t +
453 0.5 * points[2].a * points[2].t * points[2].t -
454 params.maxJ * points[2].t * points[2].t * points[2].t / 6.0;
455 }

461 void profileAllStages() {
462 logger.debug("Generated profile is 7 (all) stages.");
463 const Unit absTarget{abs(target)};
464 points[0].t = points[2].t = points[4].t = points[6].t =
465 params.maxA / params.maxJ;
466 points[1].s = params.maxJ * points[0].t * points[0].t * points[0].t / 6.0;
467 points[1].v = points[6].v = 0.5 * params.maxJ * points[0].t * points[0].t;
468 points[1].a = points[2].a = params.maxA;
469 points[1].t = points[5].t = (params.maxV - points[1].v * 2.0) / params.maxA;
470 points[2].s = points[6].s = points[1].v * points[1].t +
471 0.5 * points[1].a * points[1].t * points[1].t;
472 points[2].v = points[5].v = points[1].v + points[1].a * points[1].t;
473 points[3].s = points[5].s =
474 points[2].v * points[2].t +
475 0.5 * points[2].a * points[2].t * points[2].t -
476 params.maxJ * points[2].t * points[2].t * points[2].t / 6.0;
477 points[3].v = points[4].v = params.maxV;
478 points[3].t =
479 (absTarget - 2.0 * (points[1].s + points[2].s + points[3].s)) /
480 params.maxV;
481 points[4].s = points[3].v * points[3].t;
482 points[5].a = points[6].a = -params.maxA;
483 }

490 void finishProfile() {
491 points[0].j = points[6].j = params.maxJ;
492 points[2].j = points[4].j = -params.maxJ;
493 if(target < Unit{0.0}) {
494 for(Point &p : points) {
495 p.s *= -1;
496 p.v *= -1;
497 p.a *= -1;
498 p.j *= -1;
499 }
500 }
501 points[0].s = start;
502 for(std::size_t i{1}; i < 7; i++) {
503 points[i].t += points[i - 1].t;
504 points[i].s += points[i - 1].s;
505 }
506 }
