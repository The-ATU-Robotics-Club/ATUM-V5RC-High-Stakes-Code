10template <typename Unit>
11class MotionProfile {
12 public:
13 using UnitsPerSecond = decltype(Unit{1} / 1_s);
14 using UnitsPerSecondSq = decltype(UnitsPerSecond{1} / 1_s);
15 using UnitsPerSecondCb = decltype(UnitsPerSecondSq{1} / 1_s);
16 using UnitKinematics = Kinematics<Unit>;
17
18 struct Parameters {
19 UnitsPerSecond maxV;
20 UnitsPerSecondSq maxA;
21 // Default to a trapezoidal profile.
22 UnitsPerSecondCb maxJ{std::numeric_limits<double>::max()};
23 };
24
25 struct Point {
26 Unit s;
27 UnitsPerSecond v;
28 UnitsPerSecondSq a;
29 UnitsPerSecondCb j;
30 second_t t;
31 };
32
33 MotionProfile(const Unit start,
34 const Unit iEnd,
35 const Parameters &iParams,
36 const Logger::Level &loggerLevel = Logger::Level::Info) :
37 end{iEnd},
38 params{iParams},
39 logger{loggerLevel} {
40 points[0].s = start;
41 const Unit target{end - start};
42 beginProfile(abs(target));
43 finishProfile(target);
44 }
45
46 Point getPoint(const second_t t) {
47 if(t < 0_s) {
48 return points[0];
49 } else if(t >= points[6].t) {
50 done = true;
51 return {end,
52 UnitsPerSecond{0},
53 UnitsPerSecondSq{0},
54 UnitsPerSecondCb{0},
55 points[6].t};
56 }
57 int i{0};
58 while(t >= points[i].t) {
59 i++;
60 }
61 second_t dt{t};
62 if(i > 0) {
63 dt -= points[i - 1].t;
64 }
65 Point p;
66 p.s = UnitKinematics::position(
67 dt, points[i].s, points[i].v, points[i].a, points[i].j);
68 p.v = UnitKinematics::velocity(dt, points[i].v, points[i].a, points[i].j);
69 p.a = points[i].a + points[i].j * dt;
70 p.j = points[i].j;
71 p.t = t;
72 return p;
73 }
74
75 bool isDone() const {
76 return done;
77 }
78
79 private:
80 void beginProfile(const Unit &target) {
81 const bool reachesMaxAccel{params.maxA * params.maxA / params.maxJ <
82 params.maxV};
83 second_t jerkT{params.maxA / params.maxJ};
84 if(!reachesMaxAccel) {
85 jerkT = second_t{sqrt(getValueAs<second_t>(jerkT))};
86 }
87 const Unit jerkDistance{params.maxJ * 2.0 * pow<3>(jerkT)};
88 Unit fullDistance{jerkDistance};
89 if(reachesMaxAccel) {
90 const second_t constantAccelT =
91 (params.maxV - params.maxJ * jerkT * jerkT) / params.maxA;
92 fullDistance = 2.0 * params.maxV * jerkT +
93 params.maxJ * jerkT * jerkT * constantAccelT +
94 params.maxA * constantAccelT * constantAccelT;
95 }
96 if(target < jerkDistance) {
97 profile4Stage(target);
98 } else if(!reachesMaxAccel) {
99 profile5Stage(target);
100 } else if(target < fullDistance) {
101 profile6Stage(target);
102 } else {
103 profileAllStages(target);
104 }
105 }
106
107 void profile4Stage(const Unit &target) {
108 logger.debug("Generated profile is 4 stages (trapezoidal).");
109 points[0].t = points[2].t = points[4].t = points[6].t =
110 second_t{cbrt(getValueAs<Unit>(target) /
111 getValueAs<UnitsPerSecondCb>(params.maxJ) / 2.0)};
112 points[1].s = points[0].t * points[0].t * points[0].t * params.maxJ / 6.0;
113 points[3].s = points[5].s = 0.5 * target - points[1].s;
114 points[1].v = points[2].v = points[5].v = points[6].v =
115 0.5 * params.maxJ * points[0].t * points[0].t;
116 points[3].v = points[4].v = points[1].v * 2.0;
117 points[1].a = points[2].a = params.maxJ * points[0].t;
118 points[5].a = points[6].a = -points[1].a;
119 }
120
121 void profile5Stage(const Unit &target) {
122 logger.debug("Generated profile is 5 stages.");
123 points[0].t = points[2].t = points[4].t = points[6].t =
124 sqrt(params.maxV / params.maxJ);
125 points[3].t = (target - params.maxV * points[0].t * 2.0) / params.maxV;
126 points[1].s = points[0].t * points[0].t * points[0].t * params.maxJ / 6.0;
127 points[3].s = points[5].s = params.maxV * points[0].t - points[1].s;
128 points[4].s = target - points[1].s * 2.0 - points[3].s * 2.0;
129 points[1].v = points[2].v = points[5].v = points[6].v = params.maxV / 2.0;
130 points[3].v = points[4].v = params.maxV;
131 points[1].a = points[2].a = params.maxJ * points[0].t;
132 points[5].a = points[6].a = -points[1].a;
133 }
134
135 void profile6Stage(const Unit &target) {
136 logger.debug("Generated profile is 6 stages.");
137 const UnitsPerSecond b{3.0 * params.maxA * params.maxA / params.maxJ};
138 const Unit c{2.0 * params.maxA * params.maxA * params.maxA / params.maxJ /
139 params.maxJ -
140 target};
141 const auto discriminant = sqrt(b * b - 4.0 * params.maxA * c);
142 const second_t t{max((-b + discriminant) / (2.0 * params.maxA),
143 (-b - discriminant) / (2.0 * params.maxA))};
144 points[0].t = points[2].t = points[4].t = points[6].t =
145 params.maxA / params.maxJ;
146 points[1].t = points[5].t = t;
147 points[1].a = points[2].a = params.maxJ * points[0].t;
148 points[5].a = points[6].a = -points[1].a;
149 points[1].v = points[6].v = 0.5 * params.maxJ * points[0].t * points[0].t;
150 points[2].v = points[5].v = points[1].v + params.maxA * points[1].t;
151 points[3].v = points[4].v = points[2].v + points[2].a * points[2].t -
152 0.5 * params.maxJ * points[2].t * points[2].t;
153 points[1].s = params.maxJ * points[0].t * points[0].t * points[0].t / 6.0;
154 points[2].s = points[6].s = points[1].v * points[1].t +
155 0.5 * points[1].a * points[1].t * points[1].t;
156 points[3].s = points[5].s =
157 points[2].v * points[2].t +
158 0.5 * points[2].a * points[2].t * points[2].t -
159 params.maxJ * points[2].t * points[2].t * points[2].t / 6.0;
160 }
161
162 void profileAllStages(const Unit &target) {
163 logger.debug("Generated profile is 7 (all) stages.");
164 points[0].t = params.maxA / params.maxJ;
165 points[1].s = params.maxJ * points[0].t * points[0].t * points[0].t / 6.0;
166 points[1].v = 0.5 * params.maxJ * points[0].t * points[0].t;
167 points[1].a = params.maxA;
168 points[1].t = (params.maxV - points[1].v * 2.0) / params.maxA;
169 points[2].s = points[1].v * points[1].t +
170 0.5 * points[1].a * points[1].t * points[1].t;
171 points[2].v = points[1].v + points[1].a * points[1].t;
172 points[2].a = params.maxA;
173 points[2].t = points[0].t;
174 points[3].s = points[2].v * points[2].t +
175 0.5 * points[2].a * points[2].t * points[2].t -
176 params.maxJ * points[2].t * points[2].t * points[2].t / 6.0;
177 points[3].v = params.maxV;
178 points[3].t = (target - 2.0 * (points[1].s + points[2].s + points[3].s)) /
179 params.maxV;
180 points[4].s = points[3].v * points[3].t;
181 points[4].v = params.maxV;
182 points[4].t = points[2].t;
183 points[5].s = points[3].s;
184 points[5].v = points[2].v;
185 points[5].a = -params.maxA;
186 points[5].t = points[1].t;
187 points[6].s = points[2].s;
188 points[6].v = points[1].v;
189 points[6].a = -params.maxA;
190 points[6].t = points[0].t;
191 }
192
193 void finishProfile(const Unit &target) {
194 points[0].j = points[6].j = params.maxJ;
195 points[2].j = points[4].j = -params.maxJ;
196 if(target < Unit{0.0}) {
197 for(Point &p : points) {
198 p.s *= -1;
199 p.v *= -1;
200 p.a *= -1;
201 p.j *= -1;
202 }
203 }
204 for(std::size_t i{1}; i < 7; i++) {
205 points[i].t += points[i - 1].t;
206 points[i].s += points[i - 1].s;
207 }
208 }
209
210 private:
211 const Unit end;
212 Parameters params;
213 std::array<Point, 7> points;
214 bool done{false};
215 Logger logger;
216};
