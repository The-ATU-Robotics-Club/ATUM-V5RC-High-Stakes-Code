244 Point getClosestPoint(const Unit s) {
245 // Return appropriate value if behind start or in front of end.
246 if(distance(s, start) > Unit{0}) {
247 return {start,
248 UnitsPerSecond{0},
249 UnitsPerSecondSq{0},
250 UnitsPerSecondCb{(isBackwards()) ? -params.maxJ : params.maxJ},
251 0_s};
252 } else if(distance(s, end) < Unit{0}) {
253 return {end,
254 UnitsPerSecond{0},
255 UnitsPerSecondSq{0},
256 UnitsPerSecondCb{0},
257 points[6].t};
258 }
259 const auto [t0, t2] = getTimeBounds(s);
260 const second_t t1{searchForClosestPointTime(s, t0, t2)};
261 return getPointAt(t1);
262 }
263
270 Point getPointAt(const second_t t) {
271 if(t < 0_s) {
272 return {start,
273 UnitsPerSecond{0},
274 UnitsPerSecondSq{0},
275 UnitsPerSecondCb{(isBackwards()) ? -params.maxJ : params.maxJ},
276 0_s};
277 } else if(t >= points[6].t) {
278 return {end,
279 UnitsPerSecond{0},
280 UnitsPerSecondSq{0},
281 UnitsPerSecondCb{0},
282 points[6].t};
283 }
284 int i{0};
285 while(t >= points[i].t) {
286 i++;
287 }
288 second_t dt{t};
289 if(i > 0) {
290 dt -= points[i - 1].t;
291 }
292 Point p;
293 p.s = UnitKinematics::position(
294 dt, points[i].s, points[i].v, points[i].a, points[i].j);
295 p.v = UnitKinematics::velocity(dt, points[i].v, points[i].a, points[i].j);
296 p.a = points[i].a + points[i].j * dt;
297 p.j = points[i].j;
298 p.t = t;
299 return p;
300 }

308 std::pair<second_t, second_t> getTimeBounds(const Unit s) {
309 // Finds first point we're past (starts with 1 since we know we're past the
310 // start if this ever gets called).
311 int i{1};
312 while(i < 7 && distance(s, points[i].s) < Unit{0}) {
313 i++;
314 }
315 if(i < 2) {
316 return {0_s, points[0].t};
317 } else if(i >= 7) {
318 return {points[5].t, points[6].t};
319 } else {
320 return {points[i - 2].t, points[i - 1].t};
321 }
322 }

333 second_t searchForClosestPointTime(const Unit s, second_t t0, second_t t2) {
334 for(std::size_t n{0}; n < params.searchIterations; n++) {
335 const second_t t1{(t0 + t2) / 2.0};
336 if(distance(s, getPointAt(t1).s) >= Unit{0}) {
337 t2 = t1;
338 } else {
339 t0 = t1;
340 }
341 }
342 return (t0 + t2) / 2.0;
343 }

366 void beginProfile() {
367 const bool reachesMaxAccel{params.maxA * params.maxA / params.maxJ <
368 params.maxV};
369 second_t jerkT{params.maxA / params.maxJ};
370 if(!reachesMaxAccel) {
371 jerkT = second_t{sqrt(getValueAs<second_t>(jerkT))};
372 }
373 const Unit jerkDistance{params.maxJ * 2.0 * pow<3>(jerkT)};
374 Unit fullDistance{jerkDistance};
375 if(reachesMaxAccel) {
376 const second_t constantAccelT =
377 (params.maxV - params.maxJ * jerkT * jerkT) / params.maxA;
378 fullDistance = 2.0 * params.maxV * jerkT +
379 params.maxJ * jerkT * jerkT * constantAccelT +
380 params.maxA * constantAccelT * constantAccelT;
381 }
382 const Unit absTarget{abs(target)};
383 if(absTarget < jerkDistance) {
384 profile4Stage();
385 } else if(!reachesMaxAccel) {
386 profile5Stage();
387 } else if(absTarget < fullDistance) {
388 profile6Stage();
389 } else {
390 profileAllStages();
391 }
392 }

398 void profile4Stage() {
399 logger.debug("Generated profile is 4 stages.");
400 const Unit absTarget{abs(target)};
401 points[0].t = points[2].t = points[4].t = points[6].t =
402 second_t{cbrt(getValueAs<Unit>(absTarget) /
403 getValueAs<UnitsPerSecondCb>(params.maxJ) / 2.0)};
404 points[1].s = points[0].t * points[0].t * points[0].t * params.maxJ / 6.0;
405 points[3].s = points[5].s = 0.5 * absTarget - points[1].s;
406 points[1].v = points[2].v = points[5].v = points[6].v =
407 0.5 * params.maxJ * points[0].t * points[0].t;
408 points[3].v = points[4].v = points[1].v * 2.0;
409 points[1].a = points[2].a = params.maxJ * points[0].t;
410 points[5].a = points[6].a = -points[1].a;
411 }

417 void profile5Stage() {
418 logger.debug("Generated profile is 5 stages.");
419 const Unit absTarget{abs(target)};
420 points[0].t = points[2].t = points[4].t = points[6].t =
421 sqrt(params.maxV / params.maxJ);
422 points[3].t = (absTarget - params.maxV * points[0].t * 2.0) / params.maxV;
423 points[1].s = points[0].t * points[0].t * points[0].t * params.maxJ / 6.0;
424 points[3].s = points[5].s = params.maxV * points[0].t - points[1].s;
425 points[4].s = absTarget - points[1].s * 2.0 - points[3].s * 2.0;
426 points[1].v = points[2].v = points[5].v = points[6].v = params.maxV / 2.0;
427 points[3].v = points[4].v = params.maxV;
428 points[1].a = points[2].a = params.maxJ * points[0].t;
429 points[5].a = points[6].a = -points[1].a;
430 }

436 void profile6Stage() {
437 logger.debug("Generated profile is 6 stages.");
438 const Unit absTarget{abs(target)};
439 const UnitsPerSecond b{3.0 * params.maxA * params.maxA / params.maxJ};
440 const Unit c{2.0 * params.maxA * params.maxA * params.maxA / params.maxJ /
441 params.maxJ -
442 absTarget};
443 const auto discriminant = sqrt(b * b - 4.0 * params.maxA * c);
444 const second_t t{max((-b + discriminant) / (2.0 * params.maxA),
445 (-b - discriminant) / (2.0 * params.maxA))};
446 points[0].t = points[2].t = points[4].t = points[6].t =
447 params.maxA / params.maxJ;
448 points[1].t = points[5].t = t;
449 points[1].a = points[2].a = params.maxJ * points[0].t;
450 points[5].a = points[6].a = -points[1].a;
451 points[1].v = points[6].v = 0.5 * params.maxJ * points[0].t * points[0].t;
452 points[2].v = points[5].v = points[1].v + params.maxA * points[1].t;
453 points[3].v = points[4].v = points[2].v + points[2].a * points[2].t -
454 0.5 * params.maxJ * points[2].t * points[2].t;
455 points[1].s = params.maxJ * points[0].t * points[0].t * points[0].t / 6.0;
456 points[2].s = points[6].s = points[1].v * points[1].t +
457 0.5 * points[1].a * points[1].t * points[1].t;
458 points[3].s = points[5].s =
459 points[2].v * points[2].t +
460 0.5 * points[2].a * points[2].t * points[2].t -
461 params.maxJ * points[2].t * points[2].t * points[2].t / 6.0;
462 }

468 void profileAllStages() {
469 logger.debug("Generated profile is 7 (all) stages.");
470 const Unit absTarget{abs(target)};
471 points[0].t = points[2].t = points[4].t = points[6].t =
472 params.maxA / params.maxJ;
473 points[1].s = params.maxJ * points[0].t * points[0].t * points[0].t / 6.0;
474 points[1].v = points[6].v = 0.5 * params.maxJ * points[0].t * points[0].t;
475 points[1].a = points[2].a = params.maxA;
476 points[1].t = points[5].t = (params.maxV - points[1].v * 2.0) / params.maxA;
477 points[2].s = points[6].s = points[1].v * points[1].t +
478 0.5 * points[1].a * points[1].t * points[1].t;
479 points[2].v = points[5].v = points[1].v + points[1].a * points[1].t;
480 points[3].s = points[5].s =
481 points[2].v * points[2].t +
482 0.5 * points[2].a * points[2].t * points[2].t -
483 params.maxJ * points[2].t * points[2].t * points[2].t / 6.0;
484 points[3].v = points[4].v = params.maxV;
485 points[3].t =
486 (absTarget - 2.0 * (points[1].s + points[2].s + points[3].s)) /
487 params.maxV;
488 points[4].s = points[3].v * points[3].t;
489 points[5].a = points[6].a = -params.maxA;
490 }

497 void finishProfile() {
498 points[0].j = points[6].j = params.maxJ;
499 points[2].j = points[4].j = -params.maxJ;
500 if(target < Unit{0.0}) {
501 for(Point &p : points) {
502 p.s *= -1;
503 p.v *= -1;
504 p.a *= -1;
505 p.j *= -1;
506 }
507 }
508 points[0].s = start;
509 for(std::size_t i{1}; i < 7; i++) {
510 points[i].t += points[i - 1].t;
511 points[i].s += points[i - 1].s;
512 }
513 }
